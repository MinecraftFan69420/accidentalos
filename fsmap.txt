AOS file system disk map

sector 0: boot
sector 1: file header

char signature[4] = "AOSF"
short version;
short ft_start;
short ft_sectors;
short data_start;

sectors 2-16: file table

each file table entry = 32B

char name[20];
short sector_start;
short sector_count;
char file_type = 0 | 1; (0 = file, 1 = dir)
char flags;
short parent_entry_index;
int bytes;

dir-ID 0xFFFF means root

sectors 17+: data
